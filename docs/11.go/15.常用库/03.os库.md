# os库



## 文件读写


```go
func main() {
	// contents := os.ReadFile("./1.txt")
	root, _ := os.Getwd()
	fmt.Println(root)
	// bytes, _ := ioutil.ReadFile(root + "/0506/02.os/1.txt")
	filepath := root + "/0506/02.os/1.txt"
	bytes, err := os.ReadFile(filepath)
	if err != nil {
		fmt.Println(err)
	}
	fmt.Println(string(bytes))

}
```
