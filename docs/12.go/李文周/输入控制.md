

# 交互输入
scan,scanf及scanln的区别
```go
func main(){
	var sname string;
	var age int;

	num,err :=fmt.Scan(&sname,&age);
	if err!=nil{
		fmt.Println(err);
		return;
	}
	fmt.Println(num);
	fmt.Println("sname:",sname,"age:",age);
}
```


```go
func main(){
	var sname string;
	var age int;

	num,err :=fmt.Scanf("%v %v",&sname,&age);
	if err!=nil{
		fmt.Println(err);
		return;
	}
	fmt.Println(num);
	fmt.Println("sname:",sname,"age:",age);
}
```
